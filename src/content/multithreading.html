3. Sort an array using multiple threads

import java.util.Arrays;

class SortThread extends Thread {
  int[] arr;

  SortThread(int[] arr) {
    this.arr = arr;
  }

  public void run() {
    Arrays.sort(arr);
  }
}

public class MultiThreadSort {
  public static void main(String[] args) throws InterruptedException {
    int[] arr1 = {5, 2, 8, 1};
    int[] arr2 = {9, 3, 7, 4};

    SortThread t1 = new SortThread(arr1);
    SortThread t2 = new SortThread(arr2);

    t1.start();
    t2.start();
    t1.join();
    t2.join();

    System.out.println("Sorted Array 1: " + Arrays.toString(arr1));
    System.out.println("Sorted Array 2: " + Arrays.toString(arr2));
  }
}
Output: Two sorted arrays

4. Matrix multiplication using multiple threads

class MultiplyThread extends Thread {
  int[][] a, b, result;
  int row;

  MultiplyThread(int[][] a, int[][] b, int[][] result, int row) {
    this.a = a;
    this.b = b;
    this.result = result;
    this.row = row;
  }

  public void run() {
    for (int j = 0; j < b[0].length; j++) {
      result[row][j] = 0;
      for (int k = 0; k < b.length; k++)
        result[row][j] += a[row][k] * b[k][j];
    }
  }
}

public class MatrixMultiThread {
  public static void main(String[] args) throws InterruptedException {
    int[][] a = {{1, 2}, {3, 4}};
    int[][] b = {{2, 0}, {1, 2}};
    int[][] result = new int[2][2];

    MultiplyThread[] threads = new MultiplyThread[2];
    for (int i = 0; i < 2; i++) {
      threads[i] = new MultiplyThread(a, b, result, i);
      threads[i].start();
    }

    for (MultiplyThread t : threads)
      t.join();

    for (int[] row : result) {
      for (int val : row)
        System.out.print(val + " ");
      System.out.println();
    }
  }
}
Output: Matrix product

5. Sum of all prime numbers up to a limit using threads

class PrimeSumThread extends Thread {
  int start, end, sum = 0;

  PrimeSumThread(int start, int end) {
    this.start = start;
    this.end = end;
  }

  boolean isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= Math.sqrt(n); i++)
      if (n % i == 0) return false;
    return true;
  }

  public void run() {
    for (int i = start; i <= end; i++)
      if (isPrime(i)) sum += i;
  }
}

public class PrimeSumMultiThread {
  public static void main(String[] args) throws InterruptedException {
    PrimeSumThread t1 = new PrimeSumThread(1, 50);
    PrimeSumThread t2 = new PrimeSumThread(51, 100);

    t1.start();
    t2.start();
    t1.join();
    t2.join();

    int total = t1.sum + t2.sum;
    System.out.println("Total Prime Sum: " + total);
  }
}
Output: Sum of primes from 1 to 100

6. Concurrent web crawler using threads

class WebCrawler extends Thread {
  String url;

  WebCrawler(String url) {
    this.url = url;
  }

  public void run() {
    System.out.println("Crawling: " + url);
    // Simulate crawling
    try { Thread.sleep(1000); } catch (Exception e) {}
    System.out.println("Finished: " + url);
  }
}

public class ConcurrentCrawler {
  public static void main(String[] args) {
    String[] urls = {"https://site1.com", "https://site2.com", "https://site3.com"};
    for (String url : urls)
      new WebCrawler(url).start();
  }
}
Output: Crawling multiple sites concurrently

7. Bank account with concurrent deposits and withdrawals

class Bank {
  int balance = 1000;

  synchronized void deposit(int amount) {
    balance += amount;
    System.out.println("Deposited: " + amount + ", Balance: " + balance);
  }

  synchronized void withdraw(int amount) {
    if (balance >= amount) {
      balance -= amount;
      System.out.println("Withdrawn: " + amount + ", Balance: " + balance);
    } else {
      System.out.println("Insufficient balance");
    }
  }
}

public class BankThread {
  public static void main(String[] args) {
    Bank bank = new Bank();

    Thread t1 = new Thread(() -> bank.deposit(500));
    Thread t2 = new Thread(() -> bank.withdraw(700));
    Thread t3 = new Thread(() -> bank.withdraw(900));

    t1.start(); t2.start(); t3.start();
  }
}
Output: Concurrent deposit and withdrawal operations

11. Example of synchronization keyword

class Counter {
  int count = 0;

  synchronized void increment() {
    count++;
  }
}

public class SyncExample {
  public static void main(String[] args) throws InterruptedException {
    Counter counter = new Counter();

    Thread t1 = new Thread(() -> {
      for (int i = 0; i < 1000; i++) counter.increment();
    });

    Thread t2 = new Thread(() -> {
      for (int i = 0; i < 1000; i++) counter.increment();
    });

    t1.start();
    t2.start();
    t1.join();
    t2.join();

    System.out.println("Final Count: " + counter.count);
  }
}
Output: Final Count: 2000

12. Example of notify, notifyAll, wait, sleep methods

class SharedResource {
  synchronized void waitMethod() {
    try {
      System.out.println("Waiting...");
      wait();
      System.out.println("Resumed");
    } catch (InterruptedException e) {}
  }

  synchronized void notifyMethod() {
    System.out.println("Notifying...");
    notify();
  }
}

public class WaitNotifyExample {
  public static void main(String[] args) {
    SharedResource resource = new SharedResource();

    Thread t1 = new Thread(() -> resource.waitMethod());
    Thread t2 = new Thread(() -> {
      try { Thread.sleep(1000); } catch (Exception e) {}
      resource.notifyMethod();
    });

    t1.start();
    t2.start();
  }
}
Output: Demonstrates wait and notify behavior

13. Example of join and yield methods

class JoinYieldThread extends Thread {
  public void run() {
    for (int i = 1; i <= 3; i++) {
      System.out.println(getName() + " - Count: " + i);
      Thread.yield();
    }
  }
}

public class JoinYieldExample {
  public static void main(String[] args) throws InterruptedException {
    JoinYieldThread t1 = new JoinYieldThread();
    JoinYieldThread t2 = new JoinYieldThread();

    t1.setName("Thread-A");
    t2.setName("Thread-B");

    t1.start();
    t1.join(); // Wait for t1 to finish

    t2.start();
  }
}
Output: Thread-A completes before Thread-B starts
