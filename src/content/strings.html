20. Find all subsets of a string

public class Subsets {
  public static void main(String[] args) {
    String str = "abc";
    int len = str.length();
    int total = 1 << len;

    for (int i = 0; i < total; i++) {
      String subset = "";
      for (int j = 0; j < len; j++) {
        if ((i & (1 << j)) != 0)
          subset += str.charAt(j);
      }
      System.out.println(subset);
    }
  }
}

21. Find all permutations of a string

public class StringPermutations {
  public static void permute(String str, String result) {
    if (str.length() == 0) {
      System.out.println(result);
      return;
    }

    for (int i = 0; i < str.length(); i++) {
      char ch = str.charAt(i);
      String remaining = str.substring(0, i) + str.substring(i + 1);
      permute(remaining, result + ch);
    }
  }

  public static void main(String[] args) {
    permute("abc", "");
  }
}
Output: All permutations of "abc"

28. Find longest repeating sequence in a string

public class LongestRepeating {
  public static String findLongestRepeating(String str) {
    int n = str.length();
    String result = "";
    int[][] lcs = new int[n + 1][n + 1];

    int length = 0;
    int index = 0;

    for (int i = 1; i <= n; i++) {
      for (int j = i + 1; j <= n; j++) {
        if (str.charAt(i - 1) == str.charAt(j - 1) && lcs[i - 1][j - 1] < (j - i)) {
          lcs[i][j] = lcs[i - 1][j - 1] + 1;
          if (lcs[i][j] > length) {
            length = lcs[i][j];
            index = i;
          }
        } else {
          lcs[i][j] = 0;
        }
      }
    }

    if (length > 0)
      result = str.substring(index - length, index);

    return result;
  }

  public static void main(String[] args) {
    String str = "abcabcabc";
    System.out.println("Longest Repeating Sequence: " + findLongestRepeating(str));
  }
}
Output: Longest Repeating Sequence: abc

29. Find most repeated word in a text file

import java.io.*;
import java.util.*;

public class MostRepeatedWord {
  public static void main(String[] args) throws Exception {
    BufferedReader br = new BufferedReader(new FileReader("input.txt"));
    HashMap<String, Integer> map = new HashMap<>();
    String line;

    while ((line = br.readLine()) != null) {
      String[] words = line.toLowerCase().split("\\s+");
      for (String word : words)
        map.put(word, map.getOrDefault(word, 0) + 1);
    }

    String mostRepeated = "";
    int max = 0;
    for (String word : map.keySet()) {
      if (map.get(word) > max) {
        max = map.get(word);
        mostRepeated = word;
      }
    }

    System.out.println("Most repeated word: " + mostRepeated);
    br.close();
  }
}
Output: Most repeated word from file

public class PalindromeWords {
  public static boolean isPalindrome(String word) {
    return word.equalsIgnoreCase(new StringBuilder(word).reverse().toString());
  }

  public static void main(String[] args) {
    String str = "madam racecar level noon civic deed";
    String[] words = str.split(" ");
    String smallest = "", largest = "";

    for (String word : words) {
      if (isPalindrome(word)) {
        if (smallest.equals("") || word.length() < smallest.length())
          smallest = word;
        if (largest.equals("") || word.length() > largest.length())
          largest = word;
      }
    }

    System.out.println("Smallest Palindrome: " + smallest);
    System.out.println("Largest Palindrome: " + largest);
  }
}

public class SwapCase {
  public static void main(String[] args) {
    String str = "Java123";
    StringBuilder result = new StringBuilder();

    for (char ch : str.toCharArray()) {
      if (Character.isUpperCase(ch))
        result.append(Character.toLowerCase(ch));
      else if (Character.isLowerCase(ch))
        result.append(Character.toUpperCase(ch));
      else
        result.append(ch);
    }

    System.out.println("Swapped case: " + result);
  }
}
