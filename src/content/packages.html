1. Access java.util.Date using import keyword

import java.util.Date;

public class DateImport {
  public static void main(String[] args) {
    Date date = new Date();
    System.out.println("Current Date: " + date);
  }
}
Output: Current Date: [system date]

7. Employee payroll processing using packages

// File: employee/Emp.java
package employee;
public class Emp {
  public String name;
  public int empid;
  public String category;
  public double bpay, hra, da, npay, pf, grosspay, incometax, allowance;

  public void calculate() {
    hra = bpay * 0.2;
    da = bpay * 0.1;
    pf = bpay * 0.05;
    allowance = 1000;
    grosspay = bpay + hra + da + allowance;
    incometax = grosspay * 0.1;
    npay = grosspay - pf - incometax;
  }

  public void display() {
    System.out.println("Net Pay: " + npay);
  }
}

// File: Emppay.java
import employee.Emp;

public class Emppay {
  public static void main(String[] args) {
    Emp e = new Emp();
    e.name = "Goutam";
    e.empid = 101;
    e.category = "Developer";
    e.bpay = 50000;
    e.calculate();
    e.display();
  }
}
Output: Net Pay: [calculated value]
