1. AWT Button with FlowLayout Manager

import java.awt.*;

public class FlowButton {
  public static void main(String[] args) {
    Frame f = new Frame("FlowLayout Button");
    f.setLayout(new FlowLayout());

    Button b = new Button("Click Me");
    f.add(b);

    f.setSize(300, 200);
    f.setVisible(true);
  }
}
Output: Displays a button centered using FlowLayout.

3. Multiple Buttons with BorderLayout Manager

    f.add(new Button("North"), BorderLayout.NORTH);
    f.add(new Button("South"), BorderLayout.SOUTH);
    f.add(new Button("East"), BorderLayout.EAST);
    f.add(new Button("West"), BorderLayout.WEST);
    f.add(new Button("Center"), BorderLayout.CENTER);

    f.setSize(400, 300);
    f.setVisible(true);
  }
}
Output: Displays five buttons in BorderLayout regions.

18. AWT Calculator using GridLayout Manager

// Create a simple AWT application that designs a calculator using GridLayout manager.
import java.awt.*;

public class GridCalculator {
  public static void main(String[] args) {
    Frame f = new Frame("AWT Calculator");
    f.setLayout(new GridLayout(4, 4, 5, 5));

    TextField tf = new TextField();
    f.add(tf);

    String[] buttons = {
      "7", "8", "9", "/",
      "4", "5", "6", "*",
      "1", "2", "3", "-",
      "0", ".", "=", "+"
    };

    for (String label : buttons) {
      f.add(new Button(label));
    }

    f.setSize(300, 300);
    f.setVisible(true);
  }
}
Output: Displays a calculator layout with buttons and TextField using GridLayout.
