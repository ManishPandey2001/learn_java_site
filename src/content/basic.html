20. Compute sum of first 100 prime numbers

public class SumPrimes {
  public static boolean isPrime(int n) {
    if (n <= 1) return false;
    for (int i = 2; i <= Math.sqrt(n); i++) {
      if (n % i == 0) return false;
    }
    return true;
  }

  public static void main(String[] args) {
    int count = 0, num = 2, sum = 0;
    while (count < 100) {
      if (isPrime(num)) {
        sum += num;
        count++;
      }
      num++;
    }
    System.out.println("Sum of first 100 primes: " + sum);
  }
}
Output: Sum of first 100 prime numbers

    int num = sc.nextInt();
    int original = num, reversed = 0;

137. Program to convert Roman number to an integer number

import java.util.*;

public class RomanToInteger {
  public static int romanToInt(String s) {
    Map map = Map.of(
      'I', 1, 'V', 5, 'X', 10, 'L', 50,
      'C', 100, 'D', 500, 'M', 1000
    );
    int total = 0, prev = 0;
    for (int i = s.length() - 1; i >= 0; i--) {
      int curr = map.get(s.charAt(i));
      if (curr < prev) total -= curr;
      else total += curr;
      prev = curr;
    }
    return total;
  }

  public static void main(String[] args) {
    System.out.println("Roman to Integer: " + romanToInt("XIV"));
  }
}
Output: Roman to Integer: 14
