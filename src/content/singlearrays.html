2. Frequency of each element in the array

public class FrequencyArray {
  public static void main(String[] args) {
    int[] arr = {1, 2, 2, 3, 1, 4};
    int[] freq = new int[arr.length];
    boolean[] visited = new boolean[arr.length];

    for (int i = 0; i < arr.length; i++) {
      if (!visited[i]) {
        freq[i] = 1;
        for (int j = i + 1; j < arr.length; j++) {
          if (arr[i] == arr[j]) {
            freq[i]++;
            visited[j] = true;
          }
        }
        System.out.println(arr[i] + " occurs " + freq[i] + " times");
      }
    }
  }
}
Output: Frequency of each element

}

13. Right rotate array elements

public class RightRotate {
  public static void main(String[] args) {
    int[] arr = {1, 2, 3, 4, 5};
    int last = arr[arr.length - 1];

    for (int i = arr.length - 1; i > 0; i--)
      arr[i] = arr[i - 1];

    arr[0] = last;

    for (int val : arr)
      System.out.print(val + " ");
  }
}
Output: 5 1 2 3 4

21. Remove duplicate elements from an array

public class RemoveDuplicates {
  public static void main(String[] args) {
    int[] arr = {1, 2, 2, 3, 4, 4, 5};
    int[] temp = new int[arr.length];
    int j = 0;

    for (int i = 0; i < arr.length; i++) {
      boolean found = false;
      for (int k = 0; k < j; k++) {
        if (arr[i] == temp[k]) {
          found = true;
          break;
        }
      }
      if (!found) {
        temp[j++] = arr[i];
      }
    }

    for (int i = 0; i < j; i++)
      System.out.print(temp[i] + " ");
  }
}
Output: 1 2 3 4 5

22. Print odd and even numbers from an array

public class OddEvenArray {
  public static void main(String[] args) {
    int[] arr = {10, 15, 20, 25, 30};

    System.out.println("Even numbers:");
    for (int val : arr)
      if (val % 2 == 0)
        System.out.print(val + " ");

    System.out.println("\nOdd numbers:");
    for (int val : arr)
      if (val % 2 != 0)
        System.out.print(val + " ");
  }
}
Output:
Even numbers: 10 20 30
Odd numbers: 15 25
