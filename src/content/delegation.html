11. Java Program to Create a List Box to Select Multiple Items and Display it in the Frame

import java.awt.*;
import java.awt.event.*;

public class MultiSelectList {
  public static void main(String[] args) {
    Frame f = new Frame("Multi-Select List");
    List list = new List(5, true);
    list.setBounds(50, 50, 150, 100);
    list.add("Java");
    list.add("Python");
    list.add("C++");
    list.add("JavaScript");
    list.add("PHP");

    Button b = new Button("Show Selected");
    b.setBounds(220, 50, 120, 30);
    Label l = new Label();
    l.setBounds(50, 170, 300, 30);

    b.addActionListener(e -> {
      String[] items = list.getSelectedItems();
      StringBuilder sb = new StringBuilder("Selected: ");
      for (String item : items) sb.append(item).append(" ");
      l.setText(sb.toString());
    });

    f.add(list);
    f.add(b);
    f.add(l);
    f.setSize(400, 250);
    f.setLayout(null);
    f.setVisible(true);
  }
}
Output: Displays selected items from the list box.

18. Java Program to Display Several Dots on the Screen Continuously

import java.awt.*;
import java.util.Random;

public class DotDisplay extends Frame {
  public DotDisplay() {
    setSize(400, 300);
    setVisible(true);
  }

  public void paint(Graphics g) {
    Random rand = new Random();
    for (int i = 0; i < 100; i++) {
      int x = rand.nextInt(400);
      int y = rand.nextInt(300);
      g.fillOval(x, y, 5, 5);
      try { Thread.sleep(50); } catch (Exception e) {}
    }
  }

  public static void main(String[] args) {
    new DotDisplay();
  }
}
Output: Continuously displays random dots on the screen.

19. Java Program to Validate the TextField for only entering numbers using KeyListener

import java.awt.*;
import java.awt.event.*;

public class NumberValidation extends Frame implements KeyListener {
  TextField tf;
  Label l;

  NumberValidation() {
    tf = new TextField();
    tf.setBounds(50, 50, 150, 30);
    tf.addKeyListener(this);

    l = new Label();
    l.setBounds(50, 100, 250, 30);

    add(tf);
    add(l);
    setSize(300, 200);
    setLayout(null);
    setVisible(true);
  }

  public void keyTyped(KeyEvent e) {
    char c = e.getKeyChar();
    if (!Character.isDigit(c)) {
      e.consume();
      l.setText("Only numbers allowed!");
    } else {
      l.setText("");
    }
  }

  public void keyPressed(KeyEvent e) {}
  public void keyReleased(KeyEvent e) {}

  public static void main(String[] args) {
    new NumberValidation();
  }
}
Output: Allows only numeric input in the text field.

20. Java Program to Validate the TextField for only entering email id using KeyAdapter

import java.awt.*;
import java.awt.event.*;

public class EmailValidation extends Frame {
  TextField tf;
  Label l;

  EmailValidation() {
    tf = new TextField();
    tf.setBounds(50, 50, 200, 30);
    l = new Label();
    l.setBounds(50, 100, 300, 30);

    tf.addKeyListener(new KeyAdapter() {
      public void keyReleased(KeyEvent e) {
        String text = tf.getText();
        if (text.matches("^[\\w.-]+@[\\w.-]+\\.\\w{2,}$")) {
          l.setText("Valid Email");
        } else {
          l.setText("Invalid Email");
        }
      }
    });

    add(tf);
    add(l);
    setSize(400, 200);
    setLayout(null);
    setVisible(true);
  }

  public static void main(String[] args) {
    new EmailValidation();
  }
}
Output: Validates email format as user types.
