1. Find the smallest number among three numbers

public class SmallestOfThree {
  static int findSmallest(int a, int b, int c) {
    return Math.min(a, Math.min(b, c));
  }

  public static void main(String[] args) {
    System.out.println("Smallest: " + findSmallest(10, 5, 8));
  }
}
Output: Smallest: 5

3. Display the middle character(s) of a string

public class MiddleCharacter {
  static String getMiddle(String str) {
    int len = str.length();
    if (len % 2 == 0) {
      return str.substring(len / 2 - 1, len / 2 + 1);
    } else {
      return str.substring(len / 2, len / 2 + 1);
    }
  }

  public static void main(String[] args) {
    System.out.println("Middle: " + getMiddle("hello"));
    System.out.println("Middle: " + getMiddle("worlds"));
  }
}
Output: Middle: l
Middle: rl

5. Compute future investment value

public class InvestmentValue {
  static double futureValue(double amount, double rate, int years) {
    return amount * Math.pow(1 + rate / 100, years);
  }

  public static void main(String[] args) {
    System.out.println("Future Value: " + futureValue(10000, 5, 10));
  }
}

6. Print characters between two characters (A to P)

public class CharRange {
  static void printChars(char start, char end) {
    for (char c = start; c <= end; c++) {
      System.out.print(c + " ");
    }
  }

  public static void main(String[] args) {
    printChars('A', 'P');
  }
}
Output: A B C D E F G H I J K L M N O P

7. Check whether a string is a valid password

public class PasswordValidator {
  static boolean isValid(String password) {
    return password.length() >= 8 &&
           password.matches(".*[A-Z].*") &&
           password.matches(".*[a-z].*") &&
           password.matches(".*\\d.*");
  }

  public static void main(String[] args) {
    System.out.println("Valid: " + isValid("Java1234"));
  }
}
Output: Valid: true

9. Find all twin prime numbers less than 100

public class TwinPrimes {
  static boolean isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= Math.sqrt(n); i++)
      if (n % i == 0) return false;
    return true;
  }

  public static void main(String[] args) {
    for (int i = 2; i < 100; i++) {
      if (isPrime(i) && isPrime(i + 2)) {
        System.out.println(i + " and " + (i + 2));
      }
    }
  }
}
Output: Lists twin primes like 3 and 5, 5 and 7, etc.

10. Count the number of digits equal to 2 in an integer

public class CountTwos {
  static int countTwoDigits(int num) {
    int count = 0;
    while (num > 0) {
      if (num % 10 == 2) count++;
      num /= 10;
    }
    return count;
  }

  public static void main(String[] args) {
    System.out.println("Count of 2s: " + countTwoDigits(22122));
  }
}
Output: Count of 2s: 4

11. Check whether three integers are consecutive

public class ConsecutiveCheck {
  static boolean areConsecutive(int a, int b, int c) {
    int[] arr = {a, b, c};
    java.util.Arrays.sort(arr);
    return arr[1] == arr[0] + 1 && arr[2] == arr[1] + 1;
  }

  public static void main(String[] args) {
    System.out.println("Consecutive: " + areConsecutive(3, 4, 5));
  }
}
Output: Consecutive: true

12. Check if one integer is the middle point between the other two

public class MiddlePointCheck {
  static boolean isMiddle(int a, int b, int c) {
    int[] arr = {a, b, c};
    java.util.Arrays.sort(arr);
    return arr[1] * 2 == arr[0] + arr[2];
  }

  public static void main(String[] args) {
    System.out.println("Middle point: " + isMiddle(2, 4, 6));
  }
}
Output: Middle point: true

13. Check whether all characters in a string are vowels

public class AllVowels {
  static boolean allVowels(String str) {
