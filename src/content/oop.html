16. Abstract class "Shape" with subclasses for Rectangle, Circle, Triangle

abstract class Shape {
  abstract double area();
  abstract double perimeter();
}

class Rectangle extends Shape {
  double width, height;

  Rectangle(double w, double h) {
    width = w;
    height = h;
  }

  double area() {
    return width * height;
  }

  double perimeter() {
    return 2 * (width + height);
  }
}

class Circle extends Shape {
  double radius;

  Circle(double r) {
    radius = r;
  }

  double area() {
    return Math.PI * radius * radius;
  }

  double perimeter() {
    return 2 * Math.PI * radius;
  }
}

class Triangle extends Shape {
  double a, b, c;

  Triangle(double a, double b, double c) {
    this.a = a; this.b = b; this.c = c;
  }

  double area() {
    double s = (a + b + c) / 2;
    return Math.sqrt(s * (s - a) * (s - b) * (s - c));
  }

  double perimeter() {
    return a + b + c;
  }
}

public class ShapeTest {
  public static void main(String[] args) {
    Shape s1 = new Rectangle(4, 5);
    Shape s2 = new Circle(3);
    Shape s3 = new Triangle(3, 4, 5);

    System.out.println("Rectangle Area: " + s1.area());
    System.out.println("Circle Area: " + s2.area());
    System.out.println("Triangle Area: " + s3.area());
  }
}
Output: Areas of Rectangle, Circle, Triangle

18. Class "Restaurant" with menu, prices, and ratings

import java.util.HashMap;
import java.util.ArrayList;

class Restaurant {
  HashMap<String, Double> menu = new HashMap<>();
  ArrayList<Integer> ratings = new ArrayList<>();

  void addItem(String item, double price) {
    menu.put(item, price);
  }

  void removeItem(String item) {
    menu.remove(item);
  }

  void addRating(int rating) {
    ratings.add(rating);
  }

  double averageRating() {
    int sum = 0;
    for (int r : ratings) sum += r;
    return ratings.isEmpty() ? 0 : (double) sum / ratings.size();
  }

  public static void main(String[] args) {
    Restaurant r = new Restaurant();
    r.addItem("Pizza", 250);
    r.addItem("Burger", 150);
    r.addRating(4);
    r.addRating(5);
    System.out.println("Average Rating: " + r.averageRating());
  }
}
Output: Average Rating: 4.5
